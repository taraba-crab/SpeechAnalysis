# サンプリング周波数 16kHz

RATE = 16000
MS = 500

# 複数のsin波(周波数, 振幅)の線形和をサンプルデータとする。
def sample_audio():
    data = sum(sin_wave(k, RATE, MS) * a 
               for k, a 
               in [(200, 30), (500, 20), (700, 40)])
    audio = Audio(RATE, data)
    return audio

AUDIO = sample_audio()
AUDIO.plot()

# オーディオデータのクリップリスト(FRAME_LENGTHのフレーム長, STEP(ms)ずつ)
STEP = 10
AUDIO_FRAMES = np.array(list(AUDIO.each_frame(FRAME_LENGTH, STEP)))

# 最初のフレームをプロット
def plot_first_frame():
    _, plot = plt.subplots(3, 1)
    plot0, plot1, plot2 = plot
    
    # サンプルデータの最初のframe_lengthまでのフレーム
    frame1 = AUDIO_FRAMES[0]
    plot0.plot(TIMES, frame1)

    # ハミング窓の適用
    windowed = frame1 * HW
    plot0.plot(TIMES, windowed)
    
    # 高速フーリエ変換 
    n = len(windowed)
    ffted = np.fft.rfft(windowed, n)
    freq = np.fft.rfftfreq(n, d=1/RATE)
    plot1.plot(freq, np.abs(ffted))
    
    # 逆フーリエ変換
    iffted = np.fft.irfft(ffted) * np.sqrt(n)
    plot2.plot(TIMES, np.real(iffted))
    
plot_first_frame()

# 最初の5クリップの短時間フーリエ変換のプロット

def plot_stft(count):
    _, plots = plt.subplots(count, 2)
    n = len(HW)
    for _, frame, plot in zip(range(count), AUDIO_FRAMES, plots):
        plot0, plot1 = plot
        
        windowed = frame * HW
        plot0.plot(windowed)
        
        ffted = np.abs(np.fft.rfft(windowed))
        freq = np.fft.rfftfreq(n, d=1/RATE)
        plot1.plot(freq, ffted)
    
plot_stft(5)
